Return a single PostgreSQL SELECT query only. No explanation. No markdown. No text before or after.

TABLES (use exact names):
1. failed_jobs_daily  → backup_date, vbr_server, job_name, failed_object_name  [NO status column]
2. job_sessions       → job_name, backup_type, vbr_server, start_time, end_time, duration_minutes, status, failure_message, size_gb
3. protected_vms      → object_name, object_type, job_name, vbr_server, last_successful_backup, active_restore_points, size_gb
4. repositories       → repo_name, vbr_server, total_tb, used_tb, free_tb, used_pct  [NO backup_type, NO status]
5. restore_points     → object_name, vbr_server, backup_type, restore_point_date, size_gb
6. long_running_jobs  → job_name, backup_type, vbr_server, start_time, duration_hours, alert_level

EXAMPLES:
Q: recent failed jobs
A: SELECT f.backup_date,f.vbr_server,f.job_name,f.failed_object_name,p.object_type FROM failed_jobs_daily f JOIN protected_vms p ON p.object_name=f.failed_object_name WHERE f.backup_date>=CURRENT_DATE-1 ORDER BY f.vbr_server,f.job_name;

Q: failed jobs this week
A: SELECT f.backup_date,f.vbr_server,f.job_name,f.failed_object_name,p.object_type FROM failed_jobs_daily f JOIN protected_vms p ON p.object_name=f.failed_object_name WHERE f.backup_date>=CURRENT_DATE-7 ORDER BY f.backup_date DESC;

Q: VMs missing backups
A: SELECT object_name,vbr_server,last_successful_backup FROM protected_vms WHERE last_successful_backup < NOW()-INTERVAL '24 hours' ORDER BY last_successful_backup;

Q: repository capacity
A: SELECT repo_name,vbr_server,total_tb,used_tb,free_tb,used_pct FROM repositories ORDER BY used_pct DESC;

Q: long running jobs
A: SELECT job_name,vbr_server,duration_hours,alert_level FROM long_running_jobs ORDER BY duration_hours DESC;

Q: restore points for WAgt03
A: SELECT restore_point_date,backup_type,size_gb FROM restore_points WHERE LOWER(object_name)=LOWER('WAgt03') ORDER BY restore_point_date DESC LIMIT 7;

Q: tell me about WApp03
A: SELECT object_name,object_type,job_name,vbr_server,last_successful_backup,active_restore_points,size_gb FROM protected_vms WHERE LOWER(object_name)=LOWER('WApp03');

Q: job sessions today
A: SELECT job_name,vbr_server,status,duration_minutes,failure_message FROM job_sessions WHERE start_time>=CURRENT_DATE ORDER BY start_time DESC;
